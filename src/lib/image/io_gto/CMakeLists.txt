INCLUDE(cxx_defaults)

SET(_target
    "io_gto"
)

SET(_sources
    init.cpp
)

ADD_LIBRARY(
  ${_target} SHARED
  ${_sources}
)
SET_TARGET_PROPERTIES(
  ${_target}
  PROPERTIES PREFIX ""
)

FIND_PACKAGE(Boost REQUIRED COMPONENTS program_options)

TARGET_LINK_LIBRARIES(
  ${_target}
  PRIVATE IOgto Boost::program_options
)

RV_STAGE(TYPE "IMAGE_FORMAT" TARGET ${_target})
